#!/bin/bash
#### accepts the number of files to be generated by each thread
numoffiles=$1
runid=`date +%H%M%d`
touch ./awsls/ebay_aws_ls$runid.sh
chmod +x ./awsls/ebay_aws_ls$runid.sh
mkdir ./files$runid
mkdir ./logs$runid
echo 'numoffiles is ' $numoffiles
echo "Runid is:" $runid
###./ebay.sh $numoffiles $threadcount $runid> ./logs$runid/ebay$threadcount$$.out 2> ./logs$runid/ebay$threadcount$$.out &
for (( i=1; i <= $numoffiles; i++ ))
do
myRandom=$runid$i
echo 'random number is ' $myRandom
filename=EBAY_dropshiporderrequest_ORD"$myRandom"_v1.xml
echo $filename
sed "s/CUSTOMERORDER/ORD$myRandom/" EBAY_dropshiporderrequest_1order1line.xml > $filename
mv $filename files$runid
done
aws s3 cp files$runid s3://b2b-oc-dropship-inbound-performance/ --recursive --profile PreProduction
